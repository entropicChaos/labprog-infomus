public class SoluzioneConFrazione {
    public static Frazione fc(int n, int[] a) {
        if (n == a.length - 1) return new Frazione(a[n], 1);
        return new Frazione(a[n], 1).piu(fc(n + 1, a).inverti());
    }
    public static void main(String[] args) {
        int[] a = new int[args.length];
        for (int i = 0; i < args.length; i++) a[i] = Integer.parseInt(args[i]);
        System.out.println(fc(0, a));
    }
}
