import java.util.Scanner;

public class SoluzioneIterativa {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);
        String prima = s.next();
        String seconda = s.next();
        int lunPrima = prima.length();
        int lunSeconda = seconda.length();
        int maxLun = lunPrima > lunSeconda ? lunPrima : lunSeconda;
        for (int i = 0; i < maxLun; i++) {
            if (i < lunPrima) System.out.print(prima.charAt(i));
            if (i < lunSeconda) System.out.print(seconda.charAt(i));
        }
        System.out.println();
    }
}
